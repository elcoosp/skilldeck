# SkillDeck v1 Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build SkillDeck v1 — a local-first Tauri desktop application providing an AI agent chat interface with branching conversations, a Superpowers-compatible skill system, MCP server discovery, three workflow patterns (sequential/parallel/evaluator-optimizer), and a three-panel React UI.

**Architecture:** Rust library crate (`skilldeck-core`) owns all business logic with zero Tauri dependency; a thin Tauri shell bridges OS integration and IPC; a React 19 + TypeScript frontend is a pure view layer communicating exclusively via `invoke()` and Tauri event subscriptions. All structured data sent to LLMs is TOON-encoded for ~40% token savings.

**Tech Stack:** Tauri 2, Rust (tokio, sea-orm 2, petgraph, dashmap, notify, reqwest), React 19, TypeScript, Vite 7, shadcn/ui, Zustand, TanStack Query, @xyflow/react, Vitest, SeaORM 2 / SQLite.

```rust
use thiserror::Error;

#[derive(Debug, Error)]
pub enum CoreError {
    #[error("Model error: {0}")]
    Model(String),
    #[error("MCP error: {0}")]
    Mcp(String),
    #[error("Skill error: {0}")]
    Skill(String),
    #[error("Database error: {0}")]
    Db(String),
    #[error("Sync error: {0}")]
    Sync(String),
    #[error("Subagent error: {0}")]
    Subagent(String),
    #[error("IO error: {0}")]
    Io(String),
    #[error("Workspace error: {0}")]
    Workspace(String),
}

impl From for CoreError {
    fn from(e: std::io::Error) -> Self {
        CoreError::Io(e.to_string())
    }
}
```

```rust
// src-tauri/skilldeck-core/src/providers/claude.rs
use async_trait::async_trait;
use futures::StreamExt;
use reqwest::Client;
use serde_json::{json, Value};
use std::sync::Arc;
use crate::{CoreError, traits::*};

pub struct ClaudeProvider {
    client: Client,
    api_key: String,
    model: String,
}

impl ClaudeProvider {
    pub fn new(client: Client, api_key: String, model: String) -> Self {
        Self { client, api_key, model }
    }
}

#[async_trait]
impl ModelProvider for ClaudeProvider {
    fn id(&self) -> &str { "claude" }
    fn display_name(&self) -> &str { "Anthropic Claude" }

    async fn complete(&self, req: CompletionRequest) -> Result {
        let messages: Vec = req.messages.iter().map(|m| {
            json!({ "role": m.role, "content": m.content })
        }).collect();

        let tools: Vec = req.tools.iter().map(|t| {
            json!({
                "name": t.name,
                "description": t.description,
                "input_schema": t.input_schema
            })
        }).collect();

        let mut body = json!({
            "model": self.model,
            "max_tokens": req.model_params.max_tokens.unwrap_or(8096),
            "messages": messages,
            "stream": true
        });

        if let Some(system) = &req.system {
            body["system"] = json!(system);
        }
        if let Some(temp) = req.model_params.temperature {
            body["temperature"] = json!(temp);
        }
        if !tools.is_empty() {
            body["tools"] = json!(tools);
        }

        let response = self.client
            .post("https://api.anthropic.com/v1/messages")
            .header("x-api-key", &self.api_key)
            .header("anthropic-version", "2023-06-01")
            .header("content-type", "application/json")
            .json(&body)
            .send()
            .await
            .map_err(|e| CoreError::Model(e.to_string()))?;

        if !response.status().is_success() {
            let err_text = response.text().await.unwrap_or_default();
            return Err(CoreError::Model(format!("Claude API error: {}", err_text)));
        }

        let byte_stream = response.bytes_stream();
        let chunk_stream = byte_stream.filter_map(|res| async move {
            match res {
                Err(e) => Some(Err(CoreError::Model(e.to_string()))),
                Ok(bytes) => {
                    let text = String::from_utf8_lossy(&bytes).to_string();
                    // SSE: lines starting with "data: "
                    for line in text.lines() {
                        if let Some(data) = line.strip_prefix("data: ") {
                            if data == "[DONE]" { return None; }
                            if let Ok(v) = serde_json::from_str::(data) {
                                // content_block_delta / message_delta token
                                if let Some(delta) = v.pointer("/delta/text") {
                                    let t = delta.as_str().unwrap_or("").to_owned();
                                    if !t.is_empty() {
                                        return Some(Ok(CompletionChunk::Token(t)));
                                    }
                                }
                                // tool use start
                                if v["type"] == "content_block_start" {
                                    if let Some(tool) = v["content_block"]["type"].as_str() {
                                        if tool == "tool_use" {
                                            // handled in subsequent delta events; simplified for now
                                        }
                                    }
                                }
                                // usage on message_delta
                                if v["type"] == "message_delta" {
                                    if let Some(usage) = v.get("usage") {
                                        return Some(Ok(CompletionChunk::Done {
                                            input_tokens: 0,
                                            output_tokens: usage["output_tokens"].as_u64().unwrap_or(0) as u32,
                                            cache_read_tokens: 0,
                                            cache_write_tokens: 0,
                                        }));
                                    }
                                }
                            }
                        }
                    }
                    None
                }
            }
        });

        Ok(Box::pin(chunk_stream))
    }
}
```

```rust
        let content = content.trim_start_matches('\u{FEFF}'); // strip BOM if present
        if !content.starts_with("---") {
            return Err(CoreError::Skill(
                "SKILL.md missing frontmatter delimiter".into(),
            ));
        }
        let rest = &content[3..]; // skip first ---
        let end = rest.find("\n---").ok_or_else(|| {
            CoreError::Skill("SKILL.md missing closing frontmatter delimiter".into())
        })?;
        let frontmatter_str = &rest[..end];
        let body = rest[end + 4..].trim_start_matches('\n').to_owned();

        let manifest: serde_json::Value = serde_yaml::from_str(frontmatter_str)
            .map_err(|e| CoreError::Skill(format!("Invalid YAML frontmatter: {}", e)))?;

        let name = manifest["name"]
            .as_str()
            .ok_or_else(|| CoreError::Skill("Skill missing 'name' in frontmatter".into()))?
            .to_owned();
        let description = manifest["description"]
            .as_str()
            .unwrap_or("")
            .to_owned();

        Ok(Skill {
            name,
            description,
            content_md: body,
            manifest,
            disk_path: Some(disk_path),
            source: source.to_owned(),
        })
    }
}

#[async_trait]
impl SkillLoader for FilesystemSkillLoader {
    async fn load(&self, source: &SkillSource) -> Result {
        match source {
            SkillSource::Filesystem(path) => {
                let skill_md_path = path.join("SKILL.md");
                if !skill_md_path.exists() {
                    return Err(CoreError::Skill(format!(
                        "SKILL.md not found at {:?}", skill_md_path
                    )));
                }
                let content = tokio::fs::read_to_string(&skill_md_path).await?;
                Self::parse_skill_md(&content, path.clone(), "local")
            }
            SkillSource::Database(_) => Err(CoreError::Skill("DB skill loading not implemented here".into())),
        }
    }
}
```

```rust
    let mut skills = Vec::new();

    let mut read_dir = match tokio::fs::read_dir(dir).await {
        Ok(rd) => rd,
        Err(_) => return skills,
    };

    while let Ok(Some(entry)) = read_dir.next_entry().await {

        if path.is_dir() {
            let skill_md = path.join("SKILL.md");

                        }
                    }
                    Err(e) => tracing::warn!("Failed to read SKILL.md at {:?}: {}", skill_md, e),
                }
            }
        }
    }
```

```rust
use std::collections::HashMap;
use crate::traits::Skill;

/// Given skills from multiple source directories ordered by priority (index 0 = highest),
/// return the resolved list where duplicated names are deduplicated (first wins),
/// plus a set of shadowed names.
pub fn resolve_skills(sources: Vec)>) -> (Vec, Vec) {
    let mut seen: HashMap = HashMap::new();
    let mut resolved = Vec::new();
    let mut shadowed = Vec::new();

    for (_source_type, skills) in sources {
        for mut skill in skills {
            if seen.contains_key(&skill.name) {
                shadowed.push(skill.name.clone());
            } else {
                seen.insert(skill.name.clone(), true);
                resolved.push(skill);
            }
        }
    }

    (resolved, shadowed)
}
```

```rust
use notify::{Event, EventKind, RecursiveMode, Watcher};
use notify_debouncer_mini::{new_debouncer, DebouncedEvent};
use std::path::PathBuf;
use std::time::Duration;
use tokio::sync::mpsc::Sender;
use crate::CoreError;

pub enum SkillWatchEvent {
    Changed(PathBuf), // path to the skill directory root
}

pub fn start_watcher(
    dir: PathBuf,
    tx: Sender,
) -> Result {
    let (debounce_tx, debounce_rx) = std::sync::mpsc::channel();

    let mut debouncer = new_debouncer(Duration::from_millis(500), None, debounce_tx)
        .map_err(|e| CoreError::Io(e.to_string()))?;

    debouncer
        .watcher()
        .watch(&dir, RecursiveMode::Recursive)
        .map_err(|e| CoreError::Io(e.to_string()))?;

    // Spawn a blocking thread to relay events to the async channel
    std::thread::spawn(move || {
        for events in debounce_rx {
            if let Ok(events) = events {
                for DebouncedEvent { path, .. } in events {
                    // Only care about SKILL.md files
                    if path.file_name().map(|n| n == "SKILL.md").unwrap_or(false) {
                        if let Some(skill_dir) = path.parent().map(|p| p.to_owned()) {
                            let tx2 = tx.clone();
                            tokio::runtime::Handle::try_current()
                                .map(|h| h.spawn(async move {
                                    let _ = tx2.send(SkillWatchEvent::Changed(skill_dir)).await;
                                }))
                                .ok();
                        }
                    }
                }
            }
        }
    });

    Ok(debouncer)
}
```

```rust
// src-tauri/skilldeck-core/src/mcp/stdio_transport.rs
use async_trait::async_trait;
use serde_json::{json, Value};
use std::sync::Arc;
use tokio::io::{AsyncBufReadExt, AsyncWriteExt, BufReader};
use tokio::process::{Child, Command};
use tokio::sync::Mutex;
use crate::{CoreError, traits::*};
use super::client::*;

pub struct StdioTransport;

struct StdioSession {
    stdin: Arc<Mutex>,
    reader: Arc<Mutex<BufReader>>,
    next_id: Arc<Mutex>,
}

#[async_trait]
impl McpSessionInner for StdioSession {
    async fn call_tool(&self, name: &str, input: Value) -> Result {
        let id = {
            let mut counter = self.next_id.lock().await;
            *counter += 1;
            *counter
        };
        let req = JsonRpcRequest::new(id, "tools/call", json!({
            "name": name,
            "arguments": input
        }));
        let mut line = serde_json::to_string(&req).unwrap();
        line.push('\n');

        {
            let mut stdin = self.stdin.lock().await;
            stdin.write_all(line.as_bytes()).await.map_err(|e| CoreError::Mcp(e.to_string()))?;
        }

        let response_line = {
            let mut reader = self.reader.lock().await;
            let mut buf = String::new();
            reader.read_line(&mut buf).await.map_err(|e| CoreError::Mcp(e.to_string()))?;
            buf
        };

        let resp: JsonRpcResponse = serde_json::from_str(&response_line)
            .map_err(|e| CoreError::Mcp(format!("Bad JSON from MCP server: {}", e)))?;

        if let Some(err) = resp.error {
            return Err(CoreError::Mcp(format!("MCP error {}: {}", err.code, err.message)));
        }

        let result = resp.result.unwrap_or(Value::Null);
        let is_error = result["isError"].as_bool().unwrap_or(false);
        let content = result["content"].as_array().cloned().unwrap_or_default();

        Ok(McpCallResult { content, is_error })
    }
}

#[async_trait]
impl McpTransport for StdioTransport {
    async fn connect(&self, config: &McpServerConfig) -> Result {
        let command = config.config_json["command"]
            .as_str()
            .ok_or_else(|| CoreError::Mcp("stdio config missing 'command'".into()))?;
        let args: Vec = config.config_json["args"]
            .as_array()
            .map(|a| a.iter().filter_map(|v| v.as_str()).collect())
            .unwrap_or_default();

        let mut child = Command::new(command)
            .args(&args)
            .stdin(std::process::Stdio::piped())
            .stdout(std::process::Stdio::piped())
            .spawn()
            .map_err(|e| CoreError::Mcp(format!("Failed to spawn MCP server: {}", e)))?;

        let stdin = child.stdin.take().ok_or_else(|| CoreError::Mcp("No stdin".into()))?;
        let stdout = child.stdout.take().ok_or_else(|| CoreError::Mcp("No stdout".into()))?;

        let reader = BufReader::new(stdout);
        let session_inner = StdioSession {
            stdin: Arc::new(Mutex::new(stdin)),
            reader: Arc::new(Mutex::new(reader)),
            next_id: Arc::new(Mutex::new(0)),
        };

        // Send initialize request
        let init_req = JsonRpcRequest::new(1, "initialize", json!({
            "protocolVersion": "2024-11-05",
            "capabilities": {},
            "clientInfo": { "name": "SkillDeck", "version": "0.1.0" }
        }));
        let mut line = serde_json::to_string(&init_req).unwrap();
        line.push('\n');
        {
            let mut stdin_guard = session_inner.stdin.lock().await;
            stdin_guard.write_all(line.as_bytes()).await.map_err(|e| CoreError::Mcp(e.to_string()))?;
        }

        let init_response_line = {
            let mut reader_guard = session_inner.reader.lock().await;
            let mut buf = String::new();
            reader_guard.read_line(&mut buf).await.map_err(|e| CoreError::Mcp(e.to_string()))?;
            buf
        };

        let init_resp: JsonRpcResponse = serde_json::from_str(&init_response_line)
            .map_err(|e| CoreError::Mcp(format!("Bad initialize response: {}", e)))?;

        let tools = init_resp.result
            .map(|r| parse_tools_from_initialize(&r))
            .unwrap_or_default();

        Ok(McpSession { tools, inner: Box::new(session_inner) })
    }
}
```

```rust
// src-tauri/skilldeck-core/src/workspace/context.rs
use std::path::Path;

#[derive(Debug, Clone, Default)]
pub struct WorkspaceContext {
    pub claude_md: Option,
    pub readme_excerpt: Option,
    pub gitignore_patterns: Vec,
}

impl WorkspaceContext {
    /// Build a system prompt snippet from workspace context.
    pub fn to_system_prompt_fragment(&self) -> String {
        let mut parts = Vec::new();
        if let Some(ref claude_md) = self.claude_md {
            parts.push(format!("## Project Instructions (CLAUDE.md)\n\n{}", claude_md));
        }
        if let Some(ref readme) = self.readme_excerpt {
            parts.push(format!("## Project Overview (README)\n\n{}", readme));
        }
        parts.join("\n\n---\n\n")
    }
}

pub async fn load_workspace_context(workspace_path: &Path) -> WorkspaceContext {
    let mut ctx = WorkspaceContext::default();

    // CLAUDE.md (or CLAUDE.txt)
    for name in &["CLAUDE.md", "CLAUDE.txt"] {
        let p = workspace_path.join(name);
        if p.exists() {
            ctx.claude_md = tokio::fs::read_to_string(p).await.ok();
            break;
        }
    }

    // README (first 500 chars)
    for name in &["README.md", "README.txt", "README"] {
        let p = workspace_path.join(name);
        if p.exists() {
            if let Ok(content) = tokio::fs::read_to_string(p).await {
                ctx.readme_excerpt = Some(content.chars().take(500).collect());
                break;
            }
        }
    }

    // .gitignore patterns
    let gitignore_path = workspace_path.join(".gitignore");
    if gitignore_path.exists() {
        if let Ok(content) = tokio::fs::read_to_string(gitignore_path).await {
            ctx.gitignore_patterns = content
                .lines()
                .filter(|l| !l.is_empty() && !l.starts_with('#'))
                .map(|l| l.to_owned())
                .collect();
        }
    }

    ctx
}
```

use serde_json::Value;

/// Encode structured data for LLM consumption.
/// Uses TOON when the provider supports it, falls back to compact JSON otherwise.
/// Wraps TOON output in a ```toon code block as per spec.
pub fn encode_for_llm(data: &Value, toon_supported: bool) -> String {
    if !toon_supported {
        return serde_json::to_string(data).unwrap_or_default();
    }

    match toon_format::encode(data) {
        Ok(toon_str) => format!("```toon\n{}\n```", toon_str),
        Err(_) => {
            // TOON encoding failed (e.g., not tabular enough) — fall back to JSON
            serde_json::to_string(data).unwrap_or_default()
        }
    }
}

> 2. TOON-encodes tool schemas

```rust
// src-tauri/skilldeck-core/src/agent/loop.rs
use std::sync::Arc;
use tokio::sync::mpsc::Sender;
use uuid::Uuid;
use futures::StreamExt;
use crate::{
    CoreError,
    traits::{ModelProvider, CompletionRequest, CompletionChunk, ChatMessage, ModelParams},
    events::AgentEvent,
    agent::{built_in_tools::built_in_tool_schemas, context_builder::ContextBuilder},
};

pub struct AgentLoop {
    provider: Arc,
    event_tx: Sender,
    conversation_id: Uuid,
    messages: Vec,
    system: Option,
    model_params: ModelParams,
}

impl AgentLoop {
    pub fn new(provider: Arc, event_tx: Sender) -> Self {
        Self {
            provider,
            event_tx,
            conversation_id: Uuid::new_v4(),
            messages: Vec::new(),
            system: None,
            model_params: ModelParams::default(),
        }
    }

    pub fn with_conversation_id(mut self, id: Uuid) -> Self {
        self.conversation_id = id;
        self
    }

    pub fn with_system(mut self, system: String) -> Self {
        self.system = Some(system);
        self
    }

    pub async fn send_message(&mut self, user_text: String) -> Result {
        self.messages.push(ChatMessage {
            role: "user".into(),
            content: user_text,
        });

        let mut tools = built_in_tool_schemas();

        let req = CompletionRequest {
            messages: self.messages.clone(),
            system: self.system.clone(),
            tools,
            model_params: self.model_params.clone(),
        };

        let mut stream = self.provider.complete(req).await?;
        let mut assistant_text = String::new();

        while let Some(chunk) = stream.next().await {
            match chunk? {
                CompletionChunk::Token(text) => {
                    assistant_text.push_str(&text);
                    let _ = self.event_tx.send(AgentEvent::Token {
                        conversation_id: self.conversation_id,
                        text,
                    }).await;
                }
                CompletionChunk::ToolCall { call_id, tool_name, input_json } => {
                    // Simplified: dispatch immediately (approval logic in Tauri layer)
                    let event_id = Uuid::new_v4();
                    let _ = self.event_tx.send(AgentEvent::ToolCall {
                        tool_call_event_id: event_id,
                        server: "built-in".into(),
                        tool: tool_name.clone(),
                        input: input_json.clone(),
                    }).await;
                    // For v1 this continues; approval gating added in Tauri command layer
                }
                CompletionChunk::Done { input_tokens, output_tokens, .. } => {
                    let msg_id = Uuid::new_v4();
                    self.messages.push(ChatMessage {
                        role: "assistant".into(),
                        content: assistant_text.clone(),
                    });
                    let _ = self.event_tx.send(AgentEvent::Done {
                        conversation_id: self.conversation_id,
                        message_id: msg_id,
                        input_tokens,
                        output_tokens,
                    }).await;
                    break;
                }
            }
        }

        Ok(())
    }
}
```

```rust
// src-tauri/skilldeck-core/src/workflow/graph.rs
use petgraph::graph::{DiGraph, NodeIndex};
use petgraph::algo::toposort;
use std::collections::HashMap;
use serde::{Deserialize, Serialize};

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct StepConfig {
    pub name: String,
    pub skill_name: Option,
    pub depends_on: Vec,
}

/// Build a DAG from step configs and return execution order via topological sort.
pub fn build_execution_order(steps: &[StepConfig]) -> Result<Vec, String> {
    let mut graph: DiGraph = DiGraph::new();
    let mut name_to_node: HashMap = HashMap::new();

    // Add nodes
    for (i, step) in steps.iter().enumerate() {
        let node = graph.add_node(i);
        name_to_node.insert(step.name.clone(), node);
    }

    // Add edges (dependency → step)
    for (i, step) in steps.iter().enumerate() {
        let node = name_to_node[&step.name];
        for dep_name in &step.depends_on {
            if let Some(&dep_node) = name_to_node.get(dep_name) {
                graph.add_edge(dep_node, node, ());
            } else {
                return Err(format!("Unknown dependency '{}' in step '{}'", dep_name, step.name));
            }
        }
    }

    // Topological sort
    toposort(&graph, None)
        .map(|order| order.into_iter().map(|n| *graph.node_weight(n).unwrap()).collect())
        .map_err(|_| "Workflow has a dependency cycle".into())
}

#[cfg(test)]

            StepConfig { name: "a".into(), skill_name: None, depends_on: vec![] },

        // "a" must come before "b"
```

```rust
#[serde(rename_all = "snake_case")]
pub enum WorkflowType {
    #[default]
    Sequential,
    Parallel,
    EvaluatorOptimizer,
}

#[derive(Debug, Clone, Serialize, Deserialize, Default)]
pub struct WorkflowConfig {
    pub workflow_type: WorkflowType,
    pub steps: Vec,
    pub max_iterations: Option,
    pub quality_threshold: Option,
    pub merge_strategy: Option,
}

pub struct WorkflowExecutor {
    provider: Arc,
    event_tx: Sender,
    active: Arc<DashMap>, // workflow_id -> status
}

impl WorkflowExecutor {
    pub fn new(provider: Arc, event_tx: Sender) -> Self {
        Self {
            provider,
            event_tx,
            active: Arc::new(DashMap::new()),
        }
    }

    pub async fn execute(&self, config: WorkflowConfig) -> Result<Vec, CoreError> {
        match config.workflow_type {
            WorkflowType::Sequential => self.execute_sequential(&config).await,
            WorkflowType::Parallel => self.execute_parallel(&config).await,
            WorkflowType::EvaluatorOptimizer => self.execute_eval_opt(&config).await,
        }
    }

    async fn execute_sequential(&self, config: &WorkflowConfig) -> Result<Vec, CoreError> {
        let order = build_execution_order(&config.steps)
            .map_err(|e| CoreError::Subagent(e))?;

        let mut outputs: Vec<Option> = vec![None; config.steps.len()];

        for step_idx in order {
            let step = &config.steps[step_idx];
            let prev_output = step.depends_on.first()
                .and_then(|dep_name| {
                    config.steps.iter().position(|s| &s.name == dep_name)
                        .and_then(|i| outputs[i].clone())
                });

            let prompt = prev_output
                .map(|prev| format!("Continue from: {}", prev))
                .unwrap_or_else(|| format!("Execute step: {}", step.name));

            let (tx, mut rx) = tokio::sync::mpsc::channel(64);
            let provider = self.provider.clone();
            let mut agent = AgentLoop::new(provider, tx);
            agent.send_message(prompt).await?;

            // Collect output
            let mut step_output = String::new();
            while let Some(event) = rx.recv().await {
                match event {
                    AgentEvent::Token { text, .. } => step_output.push_str(&text),
                    AgentEvent::Done { .. } => break,
                    other => { let _ = self.event_tx.send(other).await; }
                }
            }
            outputs[step_idx] = Some(step_output);
        }

        Ok(outputs.into_iter().flatten().collect())
    }

    async fn execute_parallel(&self, config: &WorkflowConfig) -> Result<Vec, CoreError> {
        let handles: Vec = config.steps.iter().map(|step| {
            let provider = self.provider.clone();
            let (tx, mut rx) = tokio::sync::mpsc::channel(64);
            let prompt = format!("Execute step: {}", step.name);
            tokio::spawn(async move {
                let mut agent = AgentLoop::new(provider, tx);
                let _ = agent.send_message(prompt).await;
                let mut out = String::new();
                while let Some(e) = rx.recv().await {
                    match e {
                        AgentEvent::Token { text, .. } => out.push_str(&text),
                        AgentEvent::Done { .. } => break,
                        _ => {}
                    }
                }
                out
            })
        }).collect();

        let mut results = Vec::new();
        for h in handles {
            results.push(h.await.map_err(|e| CoreError::Subagent(e.to_string()))?);
        }
        Ok(results)
    }

    async fn execute_eval_opt(&self, config: &WorkflowConfig) -> Result<Vec, CoreError> {
        let max_iter = config.max_iterations.unwrap_or(3);
        let threshold = config.quality_threshold.unwrap_or(0.85);
        let generator_step = config.steps.get(0)
            .ok_or_else(|| CoreError::Subagent("EvalOpt needs at least 1 step (generator)".into()))?;
        let evaluator_step = config.steps.get(1)
            .ok_or_else(|| CoreError::Subagent("EvalOpt needs step index 1 (evaluator)".into()))?;

        let mut last_output = String::new();
        let mut last_score = 0.0_f32;

        for iteration in 0..max_iter {
            // Generate
            let gen_prompt = if last_output.is_empty() {
                format!("Execute: {}", generator_step.name)
            } else {
                format!("Refine based on feedback: {}", last_output)
            };
            let (tx, mut rx) = tokio::sync::mpsc::channel(64);
            let mut agent = AgentLoop::new(self.provider.clone(), tx);
            agent.send_message(gen_prompt).await?;
            let mut generated = String::new();
            while let Some(e) = rx.recv().await {
                match e {
                    AgentEvent::Token { text, .. } => generated.push_str(&text),
                    AgentEvent::Done { .. } => break,
                    _ => {}
                }
            }

            // Evaluate
            let eval_prompt = format!("Evaluate this output and return a JSON object with 'score' (0.0-1.0) and 'feedback': {}", generated);
            let (etx, mut erx) = tokio::sync::mpsc::channel(64);
            let mut evaluator = AgentLoop::new(self.provider.clone(), etx);
            evaluator.send_message(eval_prompt).await?;
            let mut eval_output = String::new();
            while let Some(e) = erx.recv().await {
                match e {
                    AgentEvent::Token { text, .. } => eval_output.push_str(&text),
                    AgentEvent::Done { .. } => break,
                    _ => {}
                }
            }

            // Parse score from eval output (best effort)
            last_score = serde_json::from_str::(&eval_output)
                .ok()
                .and_then(|v| v["score"].as_f64())
                .map(|s| s as f32)
                .unwrap_or(0.0);

            last_output = format!("feedback: {}\ngenerated: {}", eval_output, generated);

            if last_score >= threshold {
                break;
            }
        }

        Ok(vec![generated_output_from(last_output)])
    }
}

fn generated_output_from(s: String) -> String {
    // Extract generated content from "generated: X" suffix
    s.split("generated: ").last().unwrap_or(&s).to_owned()
}
```

```rust
#[tauri::command]
pub async fn send_message(
    app: AppHandle,
    state: State,
    conversation_id: String,
    content: String,
) -> Result {
    let conv_id = Uuid::parse_str(&conversation_id).map_err(|e| e.to_string())?;
    let msg_id = Uuid::new_v4();
    let now = Utc::now();

    // Persist user message
    let user_msg = messages::ActiveModel {
        id: Set(msg_id),
        conversation_id: Set(conv_id),
        role: Set("user".into()),
        content: Set(content.clone()),
        created_at: Set(now.into()),
        ..Default::default()
    };
    let inserted = user_msg.insert(&*state.db).await.map_err(|e| e.to_string())?;

    // Fire agent loop in background
    let db = state.db.clone();
    let app2 = app.clone();
    tokio::spawn(async move {
        // Get API key from keychain
        let api_key = app2.keychain()
            .get_password("skilldeck", "anthropic")
            .unwrap_or_default();

        let client = reqwest::Client::new();
        let provider = std::sync::Arc::new(ClaudeProvider::new(
            client,
            api_key,
            "claude-sonnet-4-6".into(),
        ));

        let (event_tx, mut event_rx) = tokio::sync::mpsc::channel(128);
        let mut agent = AgentLoop::new(provider, event_tx)
            .with_conversation_id(conv_id);

        let _ = agent.send_message(content).await;

        // Relay events to Tauri frontend
        while let Some(event) = event_rx.recv().await {
            let event_name = match &event {
                AgentEvent::Token { .. } => "agent:token",
                AgentEvent::ToolCall { .. } => "agent:tool_call",
                AgentEvent::ToolResult { .. } => "agent:tool_result",
                AgentEvent::ApprovalRequired { .. } => "agent:approval_required",
                AgentEvent::Done { .. } => "agent:done",
                AgentEvent::Error { .. } => "agent:error",
                _ => "agent:event",
            };
            let _ = app2.emit_all(event_name, &event);
        }
    });
```

```ts
import { create } from 'zustand';

// Progressive feature unlock stages (0-4)
interface UnlockState {
  stage: number;
  conversationCount: number;
  workflowUseCount: number;
  workspaceEverOpened: boolean;
  showAllFeatures: boolean;
  incrementConversationCount: () => void;
  incrementWorkflowCount: () => void;
  setWorkspaceOpened: () => void;
  setShowAllFeatures: (v: boolean) => void;
}

function computeStage(s: Omit>): number {
  if (s.showAllFeatures) return 4;
  if (s.workspaceEverOpened) return 4;
  if (s.conversationCount >= 15 || s.workflowUseCount >= 3) return 3;
  if (s.conversationCount >= 5 || s.workflowUseCount >= 1) return 2;
  return 1;
}

export const useUnlockStore = create((set, get) => ({
  stage: 1,
  conversationCount: 0,
  workflowUseCount: 0,
  workspaceEverOpened: false,
  showAllFeatures: false,
  incrementConversationCount: () => set((s) => {
    const next = { ...s, conversationCount: s.conversationCount + 1 };
    return { ...next, stage: computeStage(next) };
  }),
  incrementWorkflowCount: () => set((s) => {
    const next = { ...s, workflowUseCount: s.workflowUseCount + 1 };
    return { ...next, stage: computeStage(next) };
  }),
  setWorkspaceOpened: () => set((s) => {
    const next = { ...s, workspaceEverOpened: true };
    return { ...next, stage: computeStage(next) };
  }),
  setShowAllFeatures: (v) => set((s) => {
    const next = { ...s, showAllFeatures: v };
    return { ...next, stage: computeStage(next) };

}));
```

```ts
// src/hooks/useAgentStream.ts
import { useEffect, useRef } from "react";
import { useQueryClient } from "@tanstack/react-query";
import { agentEvents } from "../lib/events";

// Maps conversation_id -> accumulated streaming text
const streamBuffers: Record = {};

export function useAgentStream(conversationId: string) {
  const qc = useQueryClient();

  useEffect(() => {
    let unlisten: (() => void) | null = null;
    let unlistenDone: (() => void) | null = null;

    agentEvents
      .onToken((e) => {
        if (e.conversation_id !== conversationId) return;
        if (!streamBuffers[conversationId]) streamBuffers[conversationId] = "";
        streamBuffers[conversationId] += e.text;
        // Invalidate messages query to trigger re-render with streaming text
        qc.setQueryData(
          ["messages", conversationId, "streaming"],
          streamBuffers[conversationId],
        );
      })
      .then((fn) => {
        unlisten = fn;
      });

    agentEvents
      .onDone((e) => {
        if (e.conversation_id !== conversationId) return;
        delete streamBuffers[conversationId];
        qc.setQueryData(["messages", conversationId, "streaming"], null);
        qc.invalidateQueries({ queryKey: ["messages", conversationId] });
      })
      .then((fn) => {
        unlistenDone = fn;
      });

    return () => {
      unlisten?.();
      unlistenDone?.();
    };
  }, [conversationId, qc]);

  const streamingText: string | null =
    qc.getQueryData(["messages", conversationId, "streaming"]) ?? null;
  return { streamingText };
}
```

```tsx
    qc.invalidateQueries({ queryKey: ["messages", conversationId] });

  };

  const label = branches[activeIdx]?.name
```

```rust
        .one(&*state.db)
        .await
        .map_err(|e| e.to_string())?
        .ok_or("Subagent not found")?;

    let output = session.output.clone().unwrap_or_default();
    let conv_id = session.parent_conversation_id;
    let now = Utc::now();

    // Insert merged message into parent conversation
    let merged_msg_id = Uuid::new_v4();
    let merged_msg = messages::ActiveModel {
        id: Set(merged_msg_id),
        conversation_id: Set(conv_id),
        parent_id: Set(Some(session.spawn_message_id)),
        role: Set("assistant".into()),
        content: Set(format!("[Subagent result]\n\n{}", output)),
        subagent_id: Set(Some(id)),
        created_at: Set(now.into()),
        ..Default::default()
    };
    merged_msg.insert(&*state.db).await.map_err(|e| e.to_string())?;

    // Update subagent session status
    let mut active: subagent_sessions::ActiveModel = session.into();
    active.status = Set("merged".into());
    active.merge_message_id = Set(Some(merged_msg_id));
    active.merged_at = Set(Some(now.into()));
    active.update(&*state.db).await.map_err(|e| e.to_string())?;

    app.emit_all("subagent:merged", serde_json::json!({
        "subagent_id": subagent_id,
        "message_id": merged_msg_id
    })).ok();

    Ok(())
}
```

```tsx
  const { showAllFeatures, setShowAllFeatures } = useUnlockStore();
  const [keys, setKeys] = useState<Record<string, string>>({});
  const [validating, setValidating] = useState<string | null>(null);

    setValidating(provider);

      }

      setValidating(null);
```
